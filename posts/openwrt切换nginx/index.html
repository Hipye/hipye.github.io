Openwrt切换nginx

2023-02-25  • 共 106 个字 • 预计阅读时间 1 分钟

nginx -t -c /etc/nginx/uci.conf #测试配置文件 nginx -T -c /etc/nginx/uci.conf #详细测试配置文件

/etc/nginx/conf.d 目录下新建subweb.conf

server {
        listen 10088;
        listen [::]:10088;
        server_name  _;
        charset utf-8;
        location / {
            root /storage/data/share;
            index index.html;
            autoindex on;
            autoindex_exact_size off;
            autoindex_localtime on;
            add_header Cache-Control no-store;
            }
        }

/etc/init.d/nginx reload #重新读取配置

/etc/init.d/nginx restart #重启nginx服务

netstat -anutp |grep 10088 #查看10088端口监听
